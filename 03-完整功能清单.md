```kotlin

response.firmwareVersion       // 蓝牙固件版本
```

**蓝牙OTA流程**:
```kotlin
val dfuHandle = DfuHandle.getInstance()
dfuHandle.initCallback()
if (dfuHandle.checkFile(path)) {
    dfuHandle.start(object : DfuHandle.IOpResult {
        override fun onActionResult(type: Int, errCode: Int) {
            when (type) {
                1 -> dfuHandle.init()
                2 -> dfuHandle.sendPacket()
                3 -> dfuHandle.check()
                4 -> dfuHandle.endAndRelease()  // 升级成功
            }
        }

        override fun onProgress(percent: Int) {
            Log.i(TAG, "升级进度: $percent%")
        }
    })
}
```
